cmake_minimum_required(VERSION 3.20)
project(Zelda)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# ---------------------------------------------------------------------------
# 1. External Dependencies (CRITICAL: These must be defined first!)
# ---------------------------------------------------------------------------

# GLAD setup
# Assuming your glad files are in vendor/glad
add_library(glad vendor/glad/src/gl.c)
target_include_directories(glad PUBLIC vendor/glad/include)

# GLFW setup
# Make sure the folder name "vendor/glfw-3.4" matches your actual folder name exactly
add_subdirectory(vendor/glfw-3.4)

# ---------------------------------------------------------------------------
# 2. Engine Logic
# ---------------------------------------------------------------------------

# Core: The logic that doesn't care about the OS
add_library(engine_core STATIC
        src/engine/core/input/InputBackend.h
        src/engine/core/input/InputBackend.cpp
        src/engine/core/input/InputCodes.h
        src/engine/core/input/KeyListener.h
        src/engine/core/input/MouseListener.h
        src/engine/core/math/Vector2.h
        src/engine/core/Platform.h
        src/engine/core/Window.h
)
# This allows "#include <core/Window.h>" to work everywhere
target_include_directories(engine_core PUBLIC src/engine)

# Platform: The logic that talks to GLFW/OpenGL
add_library(engine_platform_glfw STATIC
        src/engine/platform/glfw/input/InputCodesGLFW.h
        src/engine/platform/glfw/input/InputCodesGLFW.cpp
        src/engine/platform/glfw/input/KeyListenerGLFW.cpp
        src/engine/platform/glfw/input/MouseListenerGLFW.cpp
        src/engine/platform/glfw/PlatformGLFW.cpp
        src/engine/platform/glfw/WindowGLFW.cpp
)
# We link PUBLIC here so anyone using the platform also gets Core/GLFW/GLAD headers
target_link_libraries(engine_platform_glfw PUBLIC engine_core glfw glad)
target_include_directories(engine_platform_glfw PUBLIC src/engine)

# ---------------------------------------------------------------------------
# 3. The Unified "GasEngine" Interface
# ---------------------------------------------------------------------------

add_library(GasEngine INTERFACE)

target_link_libraries(GasEngine INTERFACE
        engine_core
        engine_platform_glfw
)

target_include_directories(GasEngine INTERFACE
        ${CMAKE_CURRENT_SOURCE_DIR}/src/engine
)

# ---------------------------------------------------------------------------
# 4. The Game Executable
# ---------------------------------------------------------------------------

add_executable(${PROJECT_NAME} src/main.cpp)

# Link to the interface we just built
target_link_libraries(${PROJECT_NAME} PRIVATE GasEngine)
