cmake_minimum_required(VERSION 4.1.1)

project(Zelda)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF) 

# GLFW backend
add_library(platform_glfw
    src/platform/glfw/WindowGLFW.cpp
    src/platform/glfw/PlatformGLFW.cpp
    src/platform/glfw/input/KeyListenerGLFW.cpp
    src/platform/glfw/input/MouseListenerGLFW.cpp
    src/platform/glfw/input/KeyCodesGLFW.cpp
)
        
target_include_directories(platform_glfw PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/src
    ${CMAKE_CURRENT_BINARY_DIR}/vendor/glfw-3.4/include
)

# Link dependencies
add_subdirectory(vendor/glfw-3.4)
target_link_libraries(platform_glfw PUBLIC glfw)

# GLAD (later this should probably be moved to rendering module)
add_library(glad vendor/glad/src/gl.c)
target_include_directories(glad PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/vendor/glad/include
)

# Executable
add_executable(${PROJECT_NAME} src/main.cpp src/core/input/InputBackend.cpp)

target_link_libraries(${PROJECT_NAME} PRIVATE
    platform_glfw
    glad
)
